res://
├─ addons/                     # (opcional) plugins de Godot
├─ autoload/                   # scripts autoload (GameState, Save, AudioBus…)
├─ assets/                     # SOLO archivos fuente (arte/sonido)
│  ├─ sprites/
│  │  ├─ player/
│  │  ├─ enemies/
│  │  ├─ tilesets/
│  │  ├─ fx/                   # explosiones, halos, partículas
│  │  └─ ui/                   # iconos, cartas, barras
│  ├─ audio/
│  │  ├─ sfx/
│  │  └─ music/
│  ├─ fonts/
│  └─ shaders/                 # .gdshader para efectos 2D (si usás)
├─ resources/                  # .tres/.res (datos editables)
│  ├─ enemies/                 # EnemyData_*.tres (vida, daño, vel)
│  ├─ weapons/                 # WeaponData_*.tres (cadencia, daño, proyectiles)
│  ├─ waves/                   # WaveConfig_*.tres (spawns por minuto/oleada)
│  └─ materials/               # (opcional) materiales 2D
├─ scenes/
│  ├─ main/                    # Main.tscn, controladores globales
│  ├─ world/                   # mapas/TileMap, decorado, colisiones
│  ├─ actors/                  # personajes con lógica (player, enemigos, aliados)
│  │  ├─ player/               # Player.tscn + Player.gd , EnemyDetector.gd
│  │  ├─ enemies/              # EnemyBasic.tscn, enemy.gd
│  │  │  ├─ Goblin/
│  │  │  │  └─ EnemyGoblin.tscn, EnemyGoblin.gd
│  │  │  └─ Wolf/              # EnemyWolf.tscn, EnemyWolf.gd, etc.
│  │  ├─ bosses/               # jefes/miniboss
│  │  └─ familiars/            # Naxu, etc.
│  ├─ components/              # Hitbox.tscn/gd, Hurtbox.tscn/gd
│  ├─ weapons/                 # escenas de armas (autodisparo, órbitas…)
│  ├─ projectiles/             # balas, cuchillas, orbes (Area2D/RigidBody2D) player_projectiles/PlayerBullet.tscn
│  ├─ pickups/                 # XP gem, monedas, cofres
│  ├─ effects/                 # escenas de partículas/animaciones
│  └─ ui/                      # HUD, menús, carta-levelup, pausa
├─ src/                        # scripts reutilizables (sin escena)
│  ├─ components/              # Projectile.gd ,Health.gd, DamageArea.gd, Knockback.gd…
│  └─ systems/                 # Spawner.tscn, spawner.gd, GameController.gd, AudioManager.gd
└─ .gitignore
